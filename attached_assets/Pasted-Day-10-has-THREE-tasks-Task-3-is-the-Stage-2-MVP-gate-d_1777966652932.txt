TASK 3 — Stage 2 MVP gate

Run the FULL pipeline programmatically and validate against 
hand-calculated values:

  cd artifacts/survey-insight-engine && python -c "
  import time
  import pandas as pd
  from src.datamap_parser import parse_datamap
  from src.raw_decoder import decode_raw_data
  from src.question_classifier import classify_questions
  from src.calculation_log import CalculationLog
  from src.single_cut import compute_single_cuts
  from src.cross_cut_engine import compute_cross_cuts
  from src.cross_cut_suggestions import suggest_cross_cuts
  from src.excel_exporter import export_single_cuts
  from src.models import (
      CrossCutSpec, AnalysisType, QuestionType
  )

  dm = parse_datamap('sample_data/Datamap_sample.xlsx')
  df, qr = decode_raw_data(
      'sample_data/Rawdata_sample.xlsx', dm)
  schema = classify_questions(
      dm, df.columns.tolist(),
      respondent_id_column='record',
      total_respondents=len(df),
      source_rawdata_path='sample_data/Rawdata_sample.xlsx'
  )
  log = CalculationLog()

  # Single cuts
  start = time.time()
  results, skips = compute_single_cuts(schema, df, log)
  print(f'Single cuts: {len(results)} results, '
        f'{time.time()-start:.2f}s')

  # Build 4 manual cross cuts of each type for validation
  ss = [q.canonical_id for q in schema.questions
        if q.question_type == QuestionType.SINGLE_SELECT
        and q.analysis_eligible and q.option_map]
  num = [q.canonical_id for q in schema.questions
         if q.question_type == QuestionType.DIRECT_NUMERIC
         and q.analysis_eligible]

  specs = []
  if len(ss) >= 2:
      specs.append(CrossCutSpec(
          cross_cut_id='CT_VALIDATION',
          title=f'{ss[0]} x {ss[1]}',
          analysis_type=AnalysisType.CROSS_TAB,
          source_question_ids=(ss[0], ss[1]),
      ))
      filter_val = list(
          schema.get_question(ss[0]).option_map.keys())[0]
      specs.append(CrossCutSpec(
          cross_cut_id='SP_VALIDATION',
          title=f'Profile of {ss[1]} within {ss[0]}',
          analysis_type=AnalysisType.SEGMENT_PROFILE,
          source_question_ids=(ss[0], ss[1]),
          filter_expr=f'{ss[0]} == {filter_val}',
      ))
  if ss and num:
      specs.append(CrossCutSpec(
          cross_cut_id='GC_VALIDATION',
          title=f'{num[0]} by {ss[0]}',
          analysis_type=AnalysisType.GROUP_COMPARISON,
          source_question_ids=(ss[0], num[0]),
      ))
  if len(num) >= 2:
      specs.append(CrossCutSpec(
          cross_cut_id='EVR_VALIDATION',
          title=f'{num[0]} vs {num[1]}',
          analysis_type=AnalysisType.EXPECTED_VS_REALIZED,
          source_question_ids=(num[0], num[1]),
      ))

  cc_results, cc_skips = compute_cross_cuts(
      specs, schema, df, log)
  print(f'Cross cuts: {len(cc_results)} results, '
        f'{len(cc_skips)} skips')

  # Validate cross-tab counts against pandas
  print('\\n=== CROSS_TAB RECONCILIATION ===')
  ct_result = next(
      (r for r in cc_results
       if r.analysis_type == AnalysisType.CROSS_TAB),
      None)
  if ct_result:
      a, b = ct_result.source_question_ids
      expected_ct = pd.crosstab(df[a], df[b], dropna=True)
      actual_ct = ct_result.result_table['counts']
      print(f'Pandas grand total: '
            f'{int(expected_ct.values.sum())}')
      print(f'Engine grand total: '
            f'{ct_result.result_table[\"grand_total\"]}')
      print(f'Match: '
            f'{int(expected_ct.values.sum()) == '
            f'ct_result.result_table[\"grand_total\"]}')

  # Validate group comparison means
  print('\\n=== GROUP_COMPARISON RECONCILIATION ===')
  gc_result = next(
      (r for r in cc_results
       if r.analysis_type == AnalysisType.GROUP_COMPARISON),
      None)
  if gc_result:
      seg_q, met_q = gc_result.source_question_ids
      met_series = pd.to_numeric(
          df[met_q], errors='coerce')
      for seg_val, seg_data in (
              gc_result.result_table['per_segment'].items()):
          mask = df[seg_q] == seg_val
          expected_mean = met_series[mask].mean()
          actual_mean = seg_data['mean']
          match = (
              abs(expected_mean - actual_mean) < 1e-10
              if pd.notna(expected_mean) else
              pd.isna(actual_mean))
          print(f'  Segment {seg_val}: '
                f'expected_mean={expected_mean:.10f} '
                f'actual_mean={actual_mean:.10f} '
                f'match={match}')

  # Export full workbook
  export_single_cuts(
      results=results,
      skips=skips,
      schema=schema,
      quality_report=qr,
      log=log,
      output_path='outputs/survey_analysis_stage2.xlsx',
      cross_cut_results=cc_results,
      cross_cut_skips=cc_skips,
  )
  import os
  size = os.path.getsize(
      'outputs/survey_analysis_stage2.xlsx')
  print(f'\\nWorkbook: '
        f'outputs/survey_analysis_stage2.xlsx '
        f'({size/1024:.1f} KB)')
  "
