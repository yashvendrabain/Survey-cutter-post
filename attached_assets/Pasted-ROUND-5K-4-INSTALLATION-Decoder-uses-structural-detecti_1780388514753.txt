files = sorted(glob.glob('outputs/*.xlsx'), key=os.path.getmtime, reverse=True)
if not files:
    print("No workbook in outputs/")
else:
    path = files[0]
    print(f"Checking: {path}\n")
    wb = openpyxl.load_workbook(path, data_only=True)
    ws_raw = wb["_RawData"]
    headers = [ws_raw.cell(row=1, column=c).value for c in range(1, ws_raw.max_column + 1)]
    
    test_cases = [
        ("Q4 (SINGLE_SELECT, must be labels)", lambda h: h == "Q4", "label_strings"),
        ("Q1 (SINGLE_SELECT regression check)", lambda h: h == "Q1", "label_strings"),
        ("Q3 multi-select binary", lambda h: h and "Q3" in str(h) and "B2B" in str(h), "selected_or_none"),
        ("Q13 grid single-select", lambda h: h and "Q13" in str(h) and "Overall" in str(h), "label_strings"),
        ("Q27 grid (THE BUG)", lambda h: h and "Q27" in str(h) and "Field" in str(h), "numeric"),
        ("Q97 grid (THE BUG)", lambda h: h and "Q97" in str(h) and "Opportunity" in str(h), "numeric"),
        ("Q14 grid rated (THE BUG)", lambda h: h and "Q14" in str(h) and "Actual" in str(h) and "Revenue" in str(h), "numeric"),
    ]
    
    print(f"{'Question':<45} {'Expected':<18} {'Top values':<40} {'PASS/FAIL'}")
    print("-" * 130)
    
    for label, matcher, expected in test_cases:
        col_idx = next((i+1 for i, h in enumerate(headers) if matcher(h)), None)
        if not col_idx:
            print(f"{label:<45} {'(not found)':<18}")
            continue
        vals = {}
        for r in range(2, ws_raw.max_row + 1):
            v = ws_raw.cell(row=r, column=col_idx).value
            key = f"{type(v).__name__}:{str(v)[:18]}"
            vals[key] = vals.get(key, 0) + 1
        top2 = sorted(vals.items(), key=lambda x: -x[1])[:2]
        top_repr = "; ".join(f"{k}({c})" for k, c in top2)
        
        # Verdict
        is_selected = any("Selected" in k for k in vals)
        is_numeric = any(k.startswith("int") or k.startswith("float") for k in vals)
        is_labels = any(k.startswith("str:") and "Selected" not in k for k in vals)
        
        if expected == "numeric":
            verdict = "PASS" if is_numeric and not is_selected else "FAIL"
        elif expected == "label_strings":
            verdict = "PASS" if is_labels and not is_selected else "FAIL"
        elif expected == "selected_or_none":
            verdict = "PASS" if is_selected or all(k.startswith("NoneType") for k in vals) else "FAIL"
        else:
            verdict = "?"
        
        print(f"{label:<45} {expected:<18} {top_repr:<40} {verdict}")
