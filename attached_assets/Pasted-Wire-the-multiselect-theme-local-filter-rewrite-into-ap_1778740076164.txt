Then inspect with this verification script:

    python3 << 'PYEOF'
    import openpyxl
    wb = openpyxl.load_workbook("outputs/survey_v3.xlsx", data_only=False)
    
    # 1. Filters sheet still exists (workbook-wide control)
    print("--- Sheets ---")
    for s in wb.worksheets:
        print(f"  {s.title!r}  state={s.sheet_state}")
    print("Filters sheet:", "Filters" in wb.sheetnames)
    
    # 2. Pick a theme sheet, verify LOCAL FILTERS row exists
    theme_sheet = None
    for s in wb.worksheets:
        if s.title not in ('_RawData','_Options','Filters','Run_Summary',
                           'Question_Metadata','Single_Cut_Index',
                           'Calculation_Log','Filter_Log','Warnings',
                           'Survey Metadata'):
            theme_sheet = s
            break
    print(f"\n--- Theme sheet: {theme_sheet.title!r} ---")
    for row in theme_sheet.iter_rows(min_row=1, max_row=10):
        for cell in row:
            if cell.value is not None:
                v = str(cell.value)[:120]
                print(f"  {cell.coordinate}: {v}")
                break
    
    # 3. Find first SUMPRODUCT formula (NOT COUNTIFS)
    print("\n--- First formula on theme sheet ---")
    found_sumproduct = False
    found_countifs = False
    for row in theme_sheet.iter_rows():
        for cell in row:
            if isinstance(cell.value, str) and 'SUMPRODUCT' in cell.value.upper():
                print(f"  {cell.coordinate}: {cell.value[:400]}")
                found_sumproduct = True
                break
            if isinstance(cell.value, str) and cell.value.upper().startswith('=COUNTIFS'):
                print(f"  WARNING — COUNTIFS still present at {cell.coordinate}!")
                found_countifs = True
        if found_sumproduct: break
    print(f"SUMPRODUCT formula found: {found_sumproduct}")
    print(f"COUNTIFS still present: {found_countifs} (should be False)")
    
    # 4. Verify a question title has a comment
    print("\n--- Question titles with comments ---")
    comment_count = 0
    for cell in theme_sheet['A']:
        if cell.comment is not None and cell.value is not None:
            print(f"  {cell.coordinate}: title={cell.value!r:60s}")
            print(f"    comment: {cell.comment.text[:100]!r}")
            comment_count += 1
            if comment_count >= 3: break
    print(f"Total cells with comments on column A: {comment_count}")
    
    # 5. Verify "wrapped" named cells exist
    print("\n--- Named cells (sample) ---")
    all_names = list(wb.defined_names.definedName) if hasattr(wb.defined_names, 'definedName') else list(wb.defined_names)
    wrapped_count = 0
    inherit_resolved_count = 0
    for n in all_names:
        nm = n.name if hasattr(n,'name') else n
        if '_wrapped' in nm:
            wrapped_count += 1
        if '_resolved' in nm:
            inherit_resolved_count += 1
    print(f"  Named cells with '_wrapped': {wrapped_count} (expect 10+)")
    print(f"  Named cells with '_resolved': {inherit_resolved_count} (expect 20+ for theme-local)")
    
    PYEOF
