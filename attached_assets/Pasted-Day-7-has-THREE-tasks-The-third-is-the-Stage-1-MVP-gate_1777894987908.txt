C. Manual reconciliation — pick 5 single-select questions, 
   3 multi-select or grid sub-rows, and 2 numeric/allocation 
   questions from the workbook. For each, compute the 
   expected value by hand (or with a one-line pandas

  Run this verification script that does it programmatically:
  
  cd artifacts/survey-insight-engine && python -c "
  import pandas as pd
  from src.datamap_parser import parse_datamap
  from src.raw_decoder import decode_raw_data
  from src.question_classifier import classify_questions
  from src.calculation_log import CalculationLog
  from src.single_cut import compute_single_cuts
  from src.models import QuestionType
  
  dm = parse_datamap('sample_data/Datamap_sample.xlsx')
  df, qr = decode_raw_data(
      'sample_data/Rawdata_sample.xlsx', dm)
  schema = classify_questions(
      dm, df.columns.tolist(),
      respondent_id_column='record',
      total_respondents=len(df),
      source_rawdata_path='sample_data/Rawdata_sample.xlsx'
  )
  log = CalculationLog()
  results, skips = compute_single_cuts(schema, df, log)
  
  print('=== SINGLE-SELECT RECONCILIATION ===')
  ss_results = [r for r in results 
                if r.question_type == QuestionType.SINGLE_SELECT
                and r.valid_n > 0][:5]
  
  for r in ss_results:
      col = r.question_id
      # Compute expected from raw data directly
      vc = df[col].value_counts(dropna=True)
      total = int(df[col].notna().sum())
      print(f'\n{col}:')
      print(f'  expected valid_n: {total}')
      print(f'  result valid_n:   {r.valid_n}')
      print(f'  match: {total == r.valid_n}')
      
      for code, payload in list(r.distribution.items())[:3]:
          expected_count = int(vc.get(code, 0))
          expected_rate = expected_count / total
          print(f'  code {code}: '
                f'expected count={expected_count} '
                f'rate={expected_rate:.10f}')
          print(f'           '
                f'actual   count={payload[\"count\"]} '
                f'rate={payload[\"rate\"]:.10f}')
          print(f'           '
                f'match: '
                f'{expected_count == payload[\"count\"]} '
                f'rate equal: '
                f'{abs(expected_rate - payload[\"rate\"]) < 1e-10}')
  
  print('\n=== NUMERIC RECONCILIATION ===')
  num_results = [r for r in results 
                 if r.question_type == QuestionType.DIRECT_NUMERIC
                 and r.valid_n > 0][:2]
  for r in num_results:
      col = r.question_id
      series = pd.to_numeric(df[col], errors='coerce')
      print(f'\n{col}:')
      print(f'  expected mean: {series.mean():.10f}')
      print(f'  actual mean:   {r.mean:.10f}')
      print(f'  match: '
            f'{abs(series.mean() - r.mean) < 1e-10}')
  "
