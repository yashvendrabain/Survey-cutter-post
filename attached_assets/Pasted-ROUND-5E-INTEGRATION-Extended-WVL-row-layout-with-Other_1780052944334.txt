import openpyxl, glob, os
files = sorted(glob.glob('outputs/*outcome*.xlsx') + glob.glob('outputs/*winners*.xlsx'), key=os.path.getmtime, reverse=True)
path = files[0]
print(f"Checking: {path}\n")

# Check 1: others_mask_data named range
wb = openpyxl.load_workbook(path, data_only=False)
nr = list(wb.defined_names)
others_named = [n for n in nr if 'others' in n.lower()]
print(f"Others-related named ranges: {others_named}")

# Check 2: Demographics sheet first single-select question — find its layout
ws = wb['Demographics']
# Find first question header
q_row = None
for r in range(1, ws.max_row + 1):
    v = ws.cell(row=r, column=1).value
    if isinstance(v, str) and ' - ' in v and v.split(' - ')[0].startswith('Q'):
        q_row = r
        break
print(f"\nFirst question on Demographics at row {q_row}: {ws.cell(row=q_row, column=1).value[:60]}")

# Show 12 rows below the header so we see grouped header, column header, first data row, and totals
print("\nRows below question header:")
for offset in range(0, 20):
    rr = q_row + offset
    items = []
    for c in range(1, 13):
        v = ws.cell(row=rr, column=c).value
        if v is not None:
            val = str(v)[:25]
            items.append(f"{ws.cell(row=rr,column=c).coordinate}={val}")
    if items:
        print(f"  R{rr}: {' | '.join(items)}")

# Check 3: read computed values to verify NUMBERS not zeros
print("\n--- Computed values check (Bug 1 sanity) ---")
data_wb = openpyxl.load_workbook(path, data_only=True)
ws_d = data_wb['Demographics']
for r in range(1, ws_d.max_row + 1):
    a = ws_d.cell(row=r, column=1).value
    if isinstance(a, str) and 'Q14' in a and ' - ' in a:
        # Find first data row by looking for numeric in column 3
        for offset in range(2, 10):
            rr = r + offset
            v3 = ws_d.cell(row=rr, column=3).value
            v5 = ws_d.cell(row=rr, column=5).value
            v7 = ws_d.cell(row=rr, column=7).value
            v9 = ws_d.cell(row=rr, column=9).value
            v11 = ws_d.cell(row=rr, column=11).value
            if isinstance(v3, (int, float)):
                print(f"  Q14 row {rr}: option={ws_d.cell(row=rr, column=2).value}, W={v3}, L={v5}, O={v7}, T={v9}, Priority={v11}")
                break
        break

# Check 4: cohort masks in _RawData
ws_raw = wb['_RawData']
hdr = [ws_raw.cell(row=1, column=c).value for c in range(1, ws_raw.max_column + 1)]
w_col = hdr.index('winners_mask_data') + 1 if 'winners_mask_data' in hdr else None
l_col = hdr.index('laggards_mask_data') + 1 if 'laggards_mask_data' in hdr else None
o_col = hdr.index('others_mask_data') + 1 if 'others_mask_data' in hdr else None
print(f"\nMask columns: w={w_col}, l={l_col}, o={o_col}")

w_count = sum(1 for r in range(2, ws_raw.max_row + 1) if ws_raw.cell(row=r, column=w_col).value in (True, 1, 'TRUE'))
l_count = sum(1 for r in range(2, ws_raw.max_row + 1) if ws_raw.cell(row=r, column=l_col).value in (True, 1, 'TRUE'))
o_count = sum(1 for r in range(2, ws_raw.max_row + 1) if ws_raw.cell(row=r, column=o_col).value in (True, 1, 'TRUE'))
total_rows = ws_raw.max_row - 1
print(f"Winner mask TRUE: {w_count}")
print(f"Laggard mask TRUE: {l_count}")
print(f"Others mask TRUE: {o_count}")
print(f"Total rows: {total_rows}")
print(f"Sum W+L+O: {w_count + l_count + o_count}")
if w_count + l_count + o_count == total_rows:
    print("PARTITION OK: every row is in exactly one cohort.")
else:
    print(f"PARTITION ISSUE: {total_rows - (w_count + l_count + o_count)} rows are in NO cohort or overlap")

# Critical: how many rows had Q18 NaN/blank?
q18_col = hdr.index('Q18') + 1 if 'Q18' in hdr else None
if q18_col:
    blank_q18 = sum(1 for r in range(2, ws_raw.max_row + 1) if ws_raw.cell(row=r, column=q18_col).value in (None, ''))
    print(f"\nRespondents with blank Q18: {blank_q18}")
    print(f"If blank_Q18 > 0 and they're in others_mask, that's the 'non-respondents as Others' bug.")
EOF
