STEP 1 — Generate a fresh workbook using sample data

Run this script:

    cd artifacts/survey-insight-engine
    python3 - << 'EOF'
    import os, sys
    sys.path.insert(0, ".")
    
    import pandas as pd
    from src.io import load_raw_data, load_datamap
    from src.datamap_parser import parse_datamap
    from src.question_classifier import classify_questions
    from src.raw_decoder import decode_raw_data
    from src.single_cut_engine import run_single_cuts
    from src.excel_exporter import export_single_cuts
    from src.ai_insights import categorize_questions_into_themes, categorize_demographic_questions
    
    # Load sample data
    raw_path = "sample_data/Rawdata_sample.xlsx"
    map_path = "sample_data/Datamap_sample.xlsx"
    
    datamap = load_datamap(map_path)
    schema = classify_questions(parse_datamap(datamap))
    decoded_df = decode_raw_data(load_raw_data(raw_path), schema)
    results, skips, log, quality_report = run_single_cuts(decoded_df, schema)
    
    # Build themes + demo priority (no AI key needed — fallback is fine)
    questions_for_themes = [
        {"question_id": q.canonical_id, "question_text": q.question_text,
         "question_type": q.question_type.value, "is_demographic": q.is_demographic}
        for q in schema.questions
    ]
    themes = categorize_questions_into_themes(questions_for_themes, cache={})
    
    demo_qs = [
        {"question_id": q.canonical_id, "question_text": q.question_text,
         "question_type": q.question_type.value}
        for q in schema.demographic_questions()
    ]
    demo_priority = categorize_demographic_questions(demo_qs, cache={})
    
    output_path = "/tmp/smoke_test.xlsx"
    export_single_cuts(
        results=results, skips=skips, schema=schema,
        quality_report=quality_report, log=log,
        output_path=output_path,
        themes=themes, decoded_df=decoded_df,
        demo_priority=demo_priority,
    )
    print(f"Workbook written to {output_path}")
    EOF

STEP 2 — Inspect the workbook with openpyxl

Run this inspection script:

    python3 - << 'EOF'
    import openpyxl
    wb = openpyxl.load_workbook("/tmp/smoke_test.xlsx", data_only=False)
    
    print("=== SHEET NAMES AND VISIBILITY ===")
    for sheet in wb.worksheets:
        state = getattr(sheet, 'sheet_state', 'visible')
        print(f"  {sheet.title!r:40s}  state={state}")
    
    print("\n=== HIDDEN SHEETS ===")
    hidden = [s.title for s in wb.worksheets if s.sheet_state == 'hidden']
    print(f"  {hidden}")
    
    print("\n=== _RawData SHEET ===")
    if '_RawData' in wb.sheetnames:
        rd = wb['_RawData']
        print(f"  Dimensions: {rd.dimensions}")
        print(f"  Row 1 (column headers, first 10):")
        for cell in list(rd.iter_rows(min_row=1, max_row=1))[0][:10]:
            print(f"    {cell.column_letter}: {cell.value}")
        print(f"  Row 2 (first respondent, first 10 cols):")
        for cell in list(rd.iter_rows(min_row=2, max_row=2))[0][:10]:
            print(f"    {cell.column_letter}: {cell.value!r}")
    else:
        print("  MISSING — not found")
    
    print("\n=== _Options SHEET ===")
    if '_Options' in wb.sheetnames:
        op = wb['_Options']
        print(f"  Dimensions: {op.dimensions}")
        print(f"  Row 1 (column headers):")
        for cell in list(op.iter_rows(min_row=1, max_row=1))[0][:10]:
            print(f"    col {cell.column_letter}: {cell.value!r}")
        print(f"  Column A values (first 10 options):")
        for cell in list(op.iter_cols(min_col=1, max_col=1))[0][:10]:
            print(f"    row {cell.row}: {cell.value!r}")
    else:
        print("  MISSING — not found")
    
    print("\n=== THEME SHEETS (visible only) ===")
    visible = [s for s in wb.worksheets if s.sheet_state == 'visible'
               and s.title not in (
                   'Run_Summary','Question_Metadata','Single_Cut_Index',
                   'Calculation_Log','Filter_Log','Warnings','Skip_Log',
               )]
    print(f"  Count: {len(visible)}")
    for sheet in visible:
        print(f"\n  Sheet: {sheet.title!r}")
        print(f"  Row 1: {sheet['A1'].value!r}")
        print(f"  Row 3 (filter label): {sheet['A3'].value!r}")
        print(f"  Row 4 col headers (first 5):")
        for cell in list(sheet.iter_rows(min_row=4, max_row=4))[0][:5]:
            if cell.value:
                print(f"    {cell.column_letter}: {cell.value!r}")
        print(f"  Row 5 (default filter values, first 5):")
        for cell in list(sheet.iter_rows(min_row=5, max_row=5))[0][:5]:
            if cell.value:
                print(f"    {cell.column_letter}: {cell.value!r}")
        
        # Find first COUNTIFS formula
        found_countifs = False
        for row in sheet.iter_rows():
            for cell in row:
                if isinstance(cell.value, str) and 'COUNTIFS' in cell.value.upper():
                    print(f"  First COUNTIFS formula at {cell.coordinate}:")
                    print(f"    {cell.value[:200]}")
                    found_countifs = True
                    break
            if found_countifs:
                break
        if not found_countifs:
            print(f"  WARNING: No COUNTIFS formula found on this sheet")
        
        # Find first AVERAGEIFS formula
        found_avg = False
        for row in sheet.iter_rows():
            for cell in row:
                if isinstance(cell.value, str) and 'AVERAGEIFS' in cell.value.upper():
                    print(f"  First AVERAGEIFS formula at {cell.coordinate}:")
                    print(f"    {cell.value[:200]}")
                    found_avg = True
                    break
            if found_avg:
                break
        
        # Check for per-question filter cells
        for row in sheet.iter_rows():
            for cell in row:
                if isinstance(cell.value, str) and 'Filter:' in cell.value:
                    print(f"  Per-question filter label at {cell.coordinate}: {cell.value!r}")
                    break
        break  # only check first theme sheet
    
    print("\n=== NAMED RANGES (first 20) ===")
    names = list(wb.defined_names.definedName)[:20]
    for dn in names:
        print(f"  {dn.name!r:30s} -> {dn.attr_text!r}")
    
    print("\n=== DATA VALIDATIONS ON FIRST THEME SHEET ===")
    if visible:
        sheet = visible[0]
        dvs = list(sheet.data_validations.dataValidation)
        print(f"  Count: {len(dvs)}")
        for dv in dvs[:5]:
            print(f"  Type={dv.type!r}  formula1={dv.formula1!r}  sqref={dv.sqRef!r}")
    
    print("\nINSPECTION COMPLETE")
    EOF
