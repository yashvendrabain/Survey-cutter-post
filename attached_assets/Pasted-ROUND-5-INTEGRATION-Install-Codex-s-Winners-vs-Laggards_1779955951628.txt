import openpyxl, glob, os
# Find the newest WVL workbook — should have a distinct filename pattern
files = sorted(glob.glob('outputs/*winners*laggards*.xlsx') + glob.glob('outputs/*WVL*.xlsx') + glob.glob('outputs/*outcome*.xlsx'), key=os.path.getmtime, reverse=True)

    files = sorted(glob.glob('outputs/*.xlsx'), key=os.path.getmtime, reverse=True)[:3]
    print(f"Couldn't auto-find WVL file. Showing newest 3 xlsx files in outputs/: {files}")
    print("Rerun this check after telling me the correct filename.")
else:
    path = files[0]
    print(f"Checking: {path}\n")
    wb = openpyxl.load_workbook(path, data_only=False)
    print(f"Sheets: {wb.sheetnames}\n")
    
    # Look for winners_mask_data and laggards_mask_data named ranges
    nr = wb.defined_names
    print(f"Named ranges:")
    for name in nr:
        if 'mask' in name.lower() or 'winner' in name.lower() or 'laggard' in name.lower():
            print(f"  {name}")
    
    # Find a single-select sheet (Demographics is likely) and check formulas
    for sheet_name in wb.sheetnames:
        if sheet_name.startswith('_') or sheet_name in ('Run_Summary', 'Question_Metadata', 'Single_Cut_Index', 'Calculation_Log', 'Filter_Log', 'Warnings'):
            continue
        ws = wb[sheet_name]
        print(f"\nSampling sheet: {sheet_name}")
        # Find first data row with content in column C (winner count)
        for r in range(1, min(ws.max_row + 1, 60)):
            v_c = ws.cell(row=r, column=3).value
            if isinstance(v_c, str) and v_c.startswith('='):
                v_d = ws.cell(row=r, column=4).value
                v_e = ws.cell(row=r, column=5).value
                v_f = ws.cell(row=r, column=6).value
                v_g = ws.cell(row=r, column=7).value
                print(f"  Row {r}:")
                print(f"    C (winner count): {str(v_c)[:80]}")
                print(f"    D (winner pct):   {str(v_d)[:80]}")
                print(f"    E (lag count):    {str(v_e)[:80]}")
                print(f"    F (lag pct):      {str(v_f)[:80]}")
                print(f"    G (delta):        {str(v_g)[:80]}")
                # Verdict
                all_formulas = all(isinstance(x, str) and x.startswith('=') for x in [v_c, v_d, v_e, v_f, v_g] if x is not None)
                print(f"    VERDICT: {'ALL FORMULAS OK' if all_formulas else 'SOME LITERALS — BUG'}")
                break
        break
