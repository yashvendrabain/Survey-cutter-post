import sys, os; sys.path.insert(0, ".")
from src.io import load_raw_data, load_datamap
from src.datamap_parser import parse_datamap
from src.question_classifier import classify_questions
from src.raw_decoder import decode_raw_data
from src.single_cut_engine import compute_single_cuts
from src.excel_exporter import export_single_cuts
from src.ai_insights import (
    categorize_questions_into_themes,
    categorize_demographic_questions,
    generate_short_labels,
)

datamap = load_datamap("sample_data/Datamap_sample.xlsx")
schema = classify_questions(parse_datamap(datamap))
decoded_df = decode_raw_data(load_raw_data("sample_data/Rawdata_sample.xlsx"), schema)
results, skips, log, qr = compute_single_cuts(decoded_df, schema)

qfs = [{"question_id": q.canonical_id, "question_text": q.question_text,
        "question_type": q.question_type.value, "is_demographic": q.is_demographic}
       for q in schema.questions]
themes = categorize_questions_into_themes(qfs, cache={})

dqs = [{"question_id": q.canonical_id, "question_text": q.question_text,
        "question_type": q.question_type.value}
       for q in schema.demographic_questions()]
demo_priority = categorize_demographic_questions(dqs, cache={})

short_labels = generate_short_labels(
    [{"question_id": q.canonical_id, "question_text": q.question_text}
     for q in schema.questions], cache={})

# Force autocalc on for the user
import openpyxl
os.makedirs("outputs", exist_ok=True)
out = "outputs/survey_v4_index_fix.xlsx"
export_single_cuts(
    results=results, skips=skips, schema=schema,
    quality_report=qr, log=log, output_path=out,
    themes=themes, decoded_df=decoded_df,
    demo_priority=demo_priority, short_labels=short_labels,
)
# Open and force auto-calc
wb = openpyxl.load_workbook(out)
wb.calculation.calcMode = "auto"
wb.calculation.fullCalcOnLoad = True
wb.calculation.forceFullCalc = True
wb.save(out)
print(f"READY: {out}")
