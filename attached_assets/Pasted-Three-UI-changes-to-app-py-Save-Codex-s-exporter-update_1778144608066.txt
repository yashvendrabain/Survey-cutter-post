  import plotly.graph_objects as go
  import plotly.express as px
  
  def _render_chart_for_single_cut(
      result,
      spec,
      display_mode: str = "Counts",
  ) -> None:
      """Render an interactive Plotly bar chart above 
      the single-cut table."""
      
      labels = []
      counts = []
      pcts = []
      flags = []
      
      for code, payload in result.distribution.items():
          labels.append(payload.get('label', str(code)))
          counts.append(payload.get('count', 0))
          pcts.append(payload.get('rate', 0) * 100)
      
      flags = _compute_outlier_flags(counts)
      
      # Color bars by flag status
      bar_colors = []
      for f in flags:
          if f == 'high':
              bar_colors.append('#CC0000')
          elif f == 'low':
              bar_colors.append('#E65100')
          else:
              bar_colors.append('#888888')
      
      if display_mode in ("Counts", "Counts + %"):
          y_values = counts
          y_label = "Count"
      else:
          y_values = pcts
          y_label = "%"
      
      fig = go.Figure(data=[
          go.Bar(
              x=labels,
              y=y_values,
              marker_color=bar_colors,
              text=[f"{c:,}" if display_mode == "Counts" 
                    else f"{p:.1f}%" if display_mode == 
                         "% only" 
                    else f"{c:,} ({p:.1f}%)"
                    for c, p in zip(counts, pcts)],
              textposition='outside',
              hovertemplate=(
                  '<b>%{x}</b><br>'
                  'Count: %{customdata[0]:,}<br>'
                  '%: %{customdata[1]:.1f}%'
                  '<extra></extra>'
              ),
              customdata=list(zip(counts, pcts)),
          )
      ])
      
      fig.update_layout(
          title={
              'text': spec.question_text[:80] + 
                      ('…' if len(spec.question_text) > 80 
                       else ''),
              'font': {
                  'family': 'Arial',
                  'size': 14,
                  'color': '#0A0A0A',
              },
          },
          xaxis={
              'title': '',
              'tickfont': {'family': 'Arial', 'size': 11},
          },
          yaxis={
              'title': y_label,
              'titlefont': {'family': 'Arial', 
                            'size': 11, 
                            'color': '#666'},
              'tickfont': {'family': 'Arial', 'size': 10},
              'gridcolor': '#F0F0F0',
          },
          plot_bgcolor='white',
          paper_bgcolor='white',
          showlegend=False,
          height=380,
          margin={'t': 50, 'b': 80, 'l': 60, 'r': 40},
          font={'family': 'Arial'},
      )
      
      st.plotly_chart(
          fig,
          use_container_width=True,
          config={
              'displayModeBar': True,
              'displaylogo': False,
              'modeBarButtonsToRemove': ['lasso2d', 
                                         'select2d'],
              'toImageButtonOptions': {
                  'format': 'png',
                  'filename': f'{spec.canonical_id}_chart',
                  'height': 600,
                  'width': 1000,
                  'scale': 2,
              },
          },
      )


  def _render_chart_for_cross_tab(
      result,
      schema,
      display_mode: str = "Counts",
  ) -> None:
      """Render a clustered bar chart for a cross-tab 
      result."""
      
      rt = result.result_table
      counts = rt.get('counts', {})
      row_labels = rt.get('row_label_map', {})
      col_labels = rt.get('column_label_map', {})
      row_pct = rt.get('row_pct', {})
      
      if not counts:
          return
      
      row_codes = sorted(counts.keys())
      col_codes = sorted({
          c for row in counts.values() 
          for c in row.keys()
      })
      
      x_categories = [
          row_labels.get(rc, str(rc)) for rc in row_codes
      ]
      
      fig = go.Figure()
      
      colors = ['#CC0000', '#0A0A0A', '#666666', 
                '#990000', '#999999', '#330000', 
                '#444444', '#FF6666']
      
      for i, cc in enumerate(col_codes):
          col_label = col_labels.get(cc, str(cc))
          if display_mode == "Row %":
              y_vals = [
                  row_pct.get(rc, {}).get(cc, 0) * 100
                  for rc in row_codes
              ]
              text_vals = [f"{v:.1f}%" for v in y_vals]
          else:
              y_vals = [
                  counts.get(rc, {}).get(cc, 0)
                  for rc in row_codes
              ]
              text_vals = [f"{v:,}" for v in y_vals]
          
          fig.add_trace(go.Bar(
              name=col_label,
              x=x_categories,
              y=y_vals,
              text=text_vals,
              textposition='outside',
              marker_color=colors[i % len(colors)],
              hovertemplate=(
                  f'<b>{col_label}</b><br>'
                  '%{x}: %{y}<extra></extra>'
              ),
          ))
      
      a, b = result.source_question_ids
      a_spec = schema.get_question(a)
      b_spec = schema.get_question(b)
      a_text = a_spec.question_text if a_spec else a
      b_text = b_spec.question_text if b_spec else b
      
      fig.update_layout(
          title={
              'text': f'{a_text[:50]}{"…" if len(a_text)>50 else ""} × {b_text[:50]}{"…" if len(b_text)>50 else ""}',
              'font': {'family': 'Arial', 'size': 13},
          },
          barmode='group',
          xaxis={
              'title': a,
              'tickfont': {'family': 'Arial', 'size': 10},
          },
          yaxis={
              'title': 'Count' if display_mode == "Counts" 
                       else '%',
              'gridcolor': '#F0F0F0',
              'tickfont': {'family': 'Arial', 'size': 10},
          },
          plot_bgcolor='white',
          paper_bgcolor='white',
          legend={
              'title': b,
              'font': {'family': 'Arial', 'size': 10},
          },
          height=440,
          margin={'t': 60, 'b': 80, 'l': 60, 'r': 40},
          font={'family': 'Arial'},
      )
      
      st.plotly_chart(
          fig,
          use_container_width=True,
          config={
              'displayModeBar': True,
              'displaylogo': False,
              'toImageButtonOptions': {
                  'format': 'png',
                  'filename': f'{result.cross_cut_id}_chart',
                  'height': 600,
                  'width': 1200,
                  'scale': 2,
              },
          },
      )
