  from src.word_survey_parser import parse_word_survey
  from collections import Counter
  
  for path, name in [
      ('sample_data/bain_ltb_survey.docx', 'Bain LTB'),
      ('sample_data/growth_agenda_survey.docx', 
       'Growth Agenda'),
  ]:
      print(f'\n=== {name} ===')
      try:
          dm = parse_word_survey(path)
          print(f'Questions parsed: '
                f'{len(dm[\"questions\"])}')
          print(f'Parser warnings: '
                f'{len(dm[\"parser_warnings\"])}')
          
          types = Counter(
              q[\"type_hint\"] for q in dm[\"questions\"])
          for t, n in sorted(types.items(), 
                              key=lambda x: str(x[0])):
              print(f'  {t}: {n}')
          
          print('First 5 questions:')
          for q in dm['questions'][:5]:
              print(f'  {q[\"canonical_id\"]:20s} '
                    f'type={q[\"type_hint\"]!s:12s} '
                    f'opts={len(q[\"options\"])} '
                    f'subs={len(q[\"sub_columns\"])}')
          
          annotations = [
              q for q in dm['questions']
              if any(w.startswith('ANNOTATION:') 
                     for w in q['warnings'])
          ]
          print(f'Annotations preserved: '
                f'{len(annotations)} questions')
      except Exception as e:
          print(f'ERROR: {type(e).__name__}: {e}')
          import traceback
          traceback.print_exc()

Old code (two separate uploaders):

New code (single drop zone):

  st.markdown(
      "Drop **any combination** of files — the tool "
      "detects what you've uploaded automatically."
  )
  
  uploaded_files = st.file_uploader(
      "Upload your survey files",
      type=["csv", "xlsx", "docx"],
      accept_multiple_files=True,
      help=(
          "**Three scenarios are supported:**\n\n"
          "1. Raw data (.csv or .xlsx) + data map (.xlsx) "
          "— two separate files\n\n"
          "2. Combined .xlsx with raw data and data map "
          "on separate sheets\n\n"
          "3. Raw data (.csv or .xlsx) + Word survey "
          "document (.docx) — tool auto-builds the data "
          "map from the Word doc"
      ),
  )
  
  # Detection preview — show what was found before run
  if uploaded_files:
      from src.io import _detect_scenario
      names = [f.name for f in uploaded_files]
      st.caption(f"Uploaded: {', '.join(names)}")
      
      # Show a brief what-was-detected line
      try:
          from src.io import _detect_scenario
          scenario = _detect_scenario(uploaded_files)
          scenario_labels = {
              "A_separate_files": 
                  "✓ Two-file input detected "
                  "(raw data + data map)",
              "B_combined_xlsx": 
                  "✓ Combined Excel detected "
                  "(data + map on separate sheets)",
              "C_word_datamap": 
                  "✓ Word survey document detected "
                  "(data map will be parsed from Word file)",
          }
          st.info(scenario_labels.get(scenario, scenario))
      except Exception:
          pass  # detection preview is informational only
