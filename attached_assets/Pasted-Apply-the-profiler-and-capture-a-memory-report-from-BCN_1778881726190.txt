import os, sys, time
os.environ["SURVEY_PROFILE_MEMORY"] = "1"
sys.path.insert(0, "/home/runner/workspace")
sys.path.insert(0, "/home/runner/workspace/artifacts/survey-insight-engine")
from src.io import load_survey_inputs
from src.question_classifier import classify_questions
from src.single_cut.engine import compute_single_cuts

# Update these paths to match your BCN file location:
BCN_PATH = "sample_data/BCN_LTB_raw_data-_3_9_26_EOD_.xlsx"
OUT_PATH = "outputs/survey_v11_bcn_profiled.xlsx"

t0 = time.time()
print(f"[{time.time()-t0:.1f}s] Loading BCN...", flush=True)
decoded_df, schema, load_report = load_survey_inputs([BCN_PATH])
print(f"[{time.time()-t0:.1f}s] Decoded: {decoded_df.shape}", flush=True)

print(f"[{time.time()-t0:.1f}s] Computing single cuts...", flush=True)
results, skips = compute_single_cuts(decoded_df, schema)
print(f"[{time.time()-t0:.1f}s] Got {len(results)} results, {len(skips)} skips", flush=True)

print(f"[{time.time()-t0:.1f}s] Exporting workbook (profiling enabled)...", flush=True)
export_single_cuts(decoded_df, schema, results, skips, OUT_PATH, load_report=load_report)
print(f"[{time.time()-t0:.1f}s] Done. Workbook at {OUT_PATH}", flush=True)
EOF
