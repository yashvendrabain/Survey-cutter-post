  def _copy_button(df: "pd.DataFrame", 
                   key: str) -> None:
      """Render a Copy Table button that copies the 
      DataFrame as TSV to clipboard. Pastes cleanly 
      into Excel preserving columns and rows."""
      
      # Build TSV string from the dataframe
      lines = []
      # Header row
      headers = list(df.columns)
      if df.index.name:
          headers = [df.index.name] + headers
      lines.append("\t".join(str(h) for h in headers))
      
      # Data rows
      for idx, row in df.iterrows():
          if df.index.name:
              row_vals = [str(idx)] + [
                  str(v) for v in row.values]
          else:
              row_vals = [str(v) for v in row.values]
          lines.append("\t".join(row_vals))
      
      tsv = "\\n".join(lines)
      
      # Escape for JS string — replace backticks and 
      # backslashes
      tsv_escaped = (tsv
          .replace('\\', '\\\\')
          .replace('`', '\\`')
          .replace('$', '\\$'))
      
      button_id = f"copy_btn_{key}"
      
      st.markdown(f"""
      <div style="display:flex;justify-content:flex-end;
        margin-bottom:4px;">
        <button 
          id="{button_id}"
          onclick="
            navigator.clipboard.writeText(`{tsv_escaped}`)
            .then(() => {{
              document.getElementById('{button_id}')
                .textContent = '✓ Copied!';
              document.getElementById('{button_id}')
                .style.background = '#2E7D32';
              setTimeout(() => {{
                document.getElementById('{button_id}')
                  .textContent = '⎘ Copy Table';
                document.getElementById('{button_id}')
                  .style.background = '#0A0A0A';
              }}, 2000);
            }})
            .catch(() => {{
              document.getElementById('{button_id}')
                .textContent = 'Copy failed';
            }});
          "
          style="
            background: #0A0A0A;
            color: white;
            border: none;
            padding: 6px 14px;
            font-family: Arial, Helvetica, sans-serif;
            font-size: 11px;
            font-weight: 600;
            letter-spacing: 0.05em;
            cursor: pointer;
            display: flex;
            align-items: center;
            gap: 6px;
            transition: background 0.15s;
          "
        >⎘ Copy Table</button>
      </div>
      """, unsafe_allow_html=True)

Call _copy_button(df, unique_key) immediately after

BUG 3 — Multi-value filter: second and third values 
get removed automatically

This is a Streamlit session_state key collision bug. 
When the user selects multiple values in a multiselect 
widget, Streamlit rerenders the script on each 
selection change. If the multiselect widget's key 
is being reset inside the rerender loop, it wipes 
the partial selection.

ROOT CAUSE: The filter row rendering loop is probably 
doing one of these:
  (a) Re-initializing the multiselect key in 
      session_state on every render
  (b) Calling st.rerun() inside an on_change callback 
      attached to the multiselect — this cuts off 
      mid-selection
  (c) The key is generated dynamically in a way that 
      changes between renders (e.g. using len() of 
      a list that grows as filters are added)

FIND all multiselect widgets in the global filter 
section and per-question filter sections. Look for 
any of these patterns and fix them:

FIX A — Do NOT reinitialize keys in session_state 
if they already exist:

  WRONG:
    st.session_state[f'gf_vals_{i}'] = []
  
  RIGHT:
    if f'gf_vals_{i}' not in st.session_state:
        st.session_state[f'gf_vals_{i}'] = []

FIX B — Remove ALL on_change or on_select callbacks 
from multiselect widgets. Never trigger a rerun 
mid-selection. Let the user finish selecting, then 
click Apply.

  WRONG:
    st.multiselect(..., on_change=some_callback)
  
  RIGHT:
    st.multiselect(...)  # no on_change
    # computation only happens when Apply button 
    # is clicked

FIX C — Use STABLE keys that don't change between 
rerenders. Never use len(list) or similar dynamic 
values in widget keys:

  WRONG:
    key=f"filter_val_{len(st.session_state['gf_filters'])}"
  
  RIGHT:
    key=f"filter_val_{i}"  # i from enumerate, stable

FIX D — For the global filter specifically, the 
correct pattern is:

  # Outside any button handler:
  selected_values = st.multiselect(
      "Values (select one or more)",
      options=available_options,
      default=st.session_state.get(
          f'gf_vals_{i}', []),
      key=f'gf_multiselect_{i}',
      # NO on_change
  )
  
  # Store the selection in session_state so it 
  # persists across rerenders:
  st.session_state[f'gf_vals_{i}'] = selected_values
  
  # Apply button reads from session_state:
  if st.button("Apply global filter"):
      vals = st.session_state.get(
          f'gf_vals_{i}', [])
      # build FilterSpec from vals
      # set pending_global_filter flag
      # st.rerun()

FIX E — For per-question filter rows with add/remove 
(the dynamic list), stable keys must use a persistent 
row ID not the current index:

  # In session_state, store each filter row with 
  # a unique ID generated at row creation time:
  
  if 'pq_filter_rows' not in st.session_state:
      st.session_state['pq_filter_rows'] = {}
  
  # When adding a row:
  import uuid
