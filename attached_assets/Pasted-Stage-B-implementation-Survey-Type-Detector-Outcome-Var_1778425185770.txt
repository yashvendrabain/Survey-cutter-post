STEP 5: ADD OUTCOME VARIABLE SELECTOR

Immediately after the survey classification banner (still before Section 3 header), add:

        st.markdown("---")
        st.markdown("### 🎯 Outcome Variable Selection")
        
        st.info("Select the primary outcome variable for segmentation analysis. Auto-selected based on survey type; you can override.")
        
        # Build dropdown options
        all_options = result.all_eligible_questions
        option_labels = [
            f"{opt.question_id}: {opt.question_text[:80]}{'...' if len(opt.question_text) > 80 else ''} (score: {opt.relevance_score:.2f})"
            for opt in all_options
        ]
        option_ids = [opt.question_id for opt in all_options]
        
        # Add "None" option
        option_labels.append("None — no outcome variable")
        option_ids.append(None)
        
        # Find current selection index
        current_outcome = st.session_state.get("outcome_variable_id")
        if current_outcome in option_ids:
            default_index = option_ids.index(current_outcome)
        else:
            default_index = 0
        
        selected_label = st.selectbox(
            "Outcome Variable",
            options=option_labels,
            index=default_index,
            help="Auto-selected based on survey type. Override if needed.",
            key="outcome_variable_selector"
        )
        
        selected_index = option_labels.index(selected_label)
        selected_id = option_ids[selected_index]
        
        # Update session state if changed
        if selected_id != st.session_state.get("outcome_variable_id"):
            st.session_state.outcome_variable_id = selected_id
            st.rerun()
        
        # Show selected outcome details
        if selected_id:
            selected_opt = next(opt for opt in all_options if opt.question_id == selected_id)
            st.success(f"**Selected:** {selected_opt.question_id} — {selected_opt.reason}")
        else:
            st.warning("No outcome variable selected. Segmentation analysis will be unavailable in later stages.")
        
        # Show top candidates
        if len(result.candidate_outcome_questions) > 1:
            with st.expander(f"📋 Top {len(result.candidate_outcome_questions)} Outcome Candidates (auto-ranked)"):
                for opt in result.candidate_outcome_questions:
                    st.markdown(f"**{opt.question_id}** — {opt.question_text}")
                    st.caption(f"Relevance: {opt.relevance_score:.2f} | {opt.reason}")
                    st.markdown("")
