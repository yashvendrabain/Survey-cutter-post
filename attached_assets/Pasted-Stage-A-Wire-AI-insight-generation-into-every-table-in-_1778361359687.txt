TASK 2 — Build the payload helper in app.py

At the top of app.py (or in a new helper section), add:

  def _build_insight_payload_single_cut(
      result, spec, filters_applied=None
  ) -> dict:
      """Build payload dict for a SingleCutResult."""
      rows = []
      if hasattr(result, 'distribution') and result.distribution:
          for code, payload in result.distribution.items():
              label = spec.option_map.get(code, str(code))
              rows.append({
                  'label': label,
                  'count': payload.get('count', 0),
                  'rate': payload.get('rate', 0.0),
              })
      
      summary = {}
      if hasattr(result, 'mean') and result.mean is not None:
          summary['mean'] = round(result.mean, 4)
      if hasattr(result, 'median') and result.median is not None:
          summary['median'] = round(result.median, 4)
      
      return {
          'table_kind': 'single_cut',
          'question_id': spec.canonical_id,
          'question_text': spec.question_text,
          'valid_n': result.valid_n,
          'missing_n': result.missing_n,
          'filters_applied': filters_applied or [],
          'rows': rows,
          'summary': summary,
      }
  
  def _build_insight_payload_cross_cut(result) -> dict:
      """Build payload dict for a CrossCutResult."""
      rt = result.result_table
      rows = []
      
      if 'counts' in rt:
          row_labels = rt.get('row_label_map', {})
          col_labels = rt.get('column_label_map', {})
          row_pcts = rt.get('row_pct', {})
          for row_code, col_dict in rt['counts'].items():
              for col_code, count in col_dict.items():
                  rows.append({
                      'row_label': row_labels.get(
                          row_code, str(row_code)),
                      'col_label': col_labels.get(
                          col_code, str(col_code)),
                      'count': count,
                      'row_pct': row_pcts.get(
                          row_code, {}).get(col_code, 0.0),
                  })
      elif 'per_segment' in rt:
          for seg_val, seg_data in rt['per_segment'].items():
              rows.append({
                  'segment_label': seg_data.get(
                      'label', str(seg_val)),
                  'n': seg_data.get('n', 0),
                  'mean': round(seg_data.get('mean', 0), 4),
                  'median': round(
                      seg_data.get('median', 0), 4),
              })
      
      return {
          'table_kind': result.analysis_type.value.lower(),
          'question_id': result.cross_cut_id,
          'question_text': result.title,
          'valid_n': rt.get('grand_total', 0),
          'missing_n': 0,
          'filters_applied': (
              [result.filter_expr] 
              if getattr(result, 'filter_expr', None) 
              else []),
          'rows': rows,
          'summary': {},
      }

  # AI insight section
  insight_key = f"insight_sc_{spec.canonical_id}"
  
  st.divider()
  
  cols = st.columns([3, 1])
  with cols[0]:
      st.markdown("**AI insight**")
  with cols[1]:
      if st.button(
          "Generate insight" if insight_key not in st.session_state 
          else "Regenerate",
          key=f"btn_{insight_key}",
          use_container_width=True,
      ):
          with st.spinner("Generating insight..."):
              from src.ai_insights import generate_insight
              # Determine if filtered or unfiltered
              filtered = st.session_state.get(
                  'filtered_results', {}).get(spec.canonical_id)
              if filtered:
                  active_result = filtered.single_cut_result
                  active_filters = [
                      _format_filter(f) 
                      for f in filtered.filters_applied
                  ]
              else:
                  active_result = result  # the unfiltered one
                  active_filters = []
              
              payload = _build_insight_payload_single_cut(
                  active_result, spec, active_filters)
              insight_result = generate_insight(
                  payload, table_kind='single_cut',
                  title_hint=spec.question_text)
              st.session_state[insight_key] = insight_result
              st.rerun()
  
  if insight_key in st.session_state:
      ir = st.session_state[insight_key]
      
      # Title in muted bold
      st.markdown(
          f"<div style='font-size: 14px; "
          f"font-weight: 500; color: #1a1a1a; "
          f"margin-bottom: 4px;'>"
          f"{ir.title}</div>",
          unsafe_allow_html=True,
      )
      
      # Insight body
      if ir.was_template:
          st.caption(
              f"📋 {ir.insight}  *(template — AI unavailable)*")
          if ir.error_message:
              with st.expander("Why was AI unavailable?"):
                  st.code(ir.error_message)
      else:
          st.markdown(
              f"<div style='font-size: 13px; "
              f"line-height: 1.7; color: #333; "
              f"background: #f8f9fa; "
              f"border-left: 3px solid #185FA5; "
              f"padding: 10px 14px; "
              f"border-radius: 0;'>"
              f"{ir.insight}</div>",
              unsafe_allow_html=True,
          )
          st.caption(
              f"Generated by {ir.model_used} · "
              f"{ir.tokens_used} tokens")
