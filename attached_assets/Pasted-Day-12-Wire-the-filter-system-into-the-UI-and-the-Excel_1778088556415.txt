  with st.expander(f"{spec.canonical_id}: {short_text}",

DISPATCH ON APPLY

When apply_filter is clicked:

  filters = []
  if demo_pick[1] is not None:
      filters.append(FilterSpec(
          filter_question_id=demo_pick[1],
          filter_value=demo_value_pick[1],
      ))
  if custom_pick[1] is not None:
      filters.append(FilterSpec(
          filter_question_id=custom_pick[1],
          filter_value=custom_value_pick[1],
      ))
  
  # Validation: max one breakdown filter (filter_value is None)
  breakdowns = [f for f in filters if f.filter_value is None]
  if len(breakdowns) > 1:
      st.error("Only one breakdown filter at a time. "
               "Either pick a value for one of them or remove it.")
  else:
      try:
          from src.filtered_single_cut import compute_filtered_single_cut
          filtered_result = compute_filtered_single_cut(
              spec.canonical_id,
              filters,
              st.session_state['schema'],
              st.session_state['decoded_df'],
              st.session_state['log'],
          )
          # Cache by spec.canonical_id
          if 'filtered_results' not in st.session_state:
              st.session_state['filtered_results'] = {}
          st.session_state['filtered_results'][spec.canonical_id] = filtered_result
          st.rerun()
      except Exception as e:
          st.error(f"Filter failed: {e}")
          import traceback
          with st.expander("Show traceback"):
              st.code(traceback.format_exc())

Helper:
  def _format_filter(f):
      if f.filter_value is None:
          return f"{f.filter_question_id} (breakdown)"
      return f"{f.filter_question_id} = {f.filter_value}"

  selected_filtered = [
      r for r in st.session_state.get('filtered_results', {}).values()
